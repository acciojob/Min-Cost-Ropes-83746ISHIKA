function mincost(arr) { // Use a min heap approach using sort let totalCost = 0; // Keep combining until only one rope is left while (arr.length > 1) { // Sort to get two smallest ropes at front arr.sort((a, b) => a - b); // Take out two smallest let first = arr.shift(); let second = arr.shift(); // Cost to combine them let cost = first + second; totalCost += cost; // Push the combined rope back arr.push(cost); } return totalCost; } // Test cases console.log(mincost([4, 3, 2, 6])); // 29 console.log(mincost([1, 2, 3, 4, 5])); // 33
